// ==UserScript== // @name Poki Name Changer naar Klas // @namespace http://tampermonkey.net // @version 1.0 // @description Verandert de naam/nickname op Poki automatisch naar 'klas' // @author Je Naam // @match https://*://* // @grant none // @run-at document-end // ==/UserScript== (function() { 'use strict'; // Functie die zoekt naar tekst/naamvelden op de pagina en deze aanpast function changeToKlas() { // Vervang deze selectors op basis van het specifieke spel of profielveld let nameFields = document.querySelectorAll('.username, .profile-name, #player-name'); nameFields.forEach(field => { field.innerText = 'klas'; field.value = 'klas'; }); } // Voer de functie direct uit en herhaal voor dynamisch ladende spellen changeToKlas(); setInterval(changeToKlas, 2000); })();
